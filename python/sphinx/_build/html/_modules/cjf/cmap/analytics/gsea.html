import matplotlib.pyplot as plt


[docs]def preranked(gene_set,ranked_list,p=1,plot=False):
    '''
    preranked gsea using gene_set and ranks
    '''
    #ensure that the inputs are lists
    gene_set = list(gene_set)
    ranked_list = list(ranked_list)
     
    gene_set_inds = list(ranked_list.index(x) for x in gene_set)
    
    #build rank_wieghts
    rank_weights = list(np.linspace(1, 0, len(ranked_list)/2))
    rank_weights.extend(list(np.linspace(0, -1, len(ranked_list)/2)))
    rank_weights = np.array(rank_weights)
    
    #set up constants
    N_r = sum(np.abs(np.array(rank_weights[gene_set_inds]))**p)
    N = len(ranked_list)
    N_h = len(gene_set)
    
    #compute the ES
    P_hit = np.cumsum(list((abs(rank_weights[x])**p)/N_r if x in gene_set_inds else 0 for x in range(N)))
    P_miss = np.cumsum(list(1.0/(N-N_h) if x not in gene_set_inds else 0 for x in range(N)))
    ES = P_hit - P_miss
    
    #plot it if required
    if plot:
        plt.plot(ES)
        plt.show()
    
    #find the maximum deviation from 0 and return it
    max_ind = np.where(np.abs(ES) == np.max(np.abs(ES)))
    return float(ES[max_ind])
