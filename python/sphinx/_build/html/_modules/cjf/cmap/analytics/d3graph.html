[docs]def build_nodes_and_links_from_table(in_file,nodes_file,links_file):
    '''
    build a set of nodes and links from an input table consisting of a 'name' column of data
    and any number of additional metadata columns consisting of 0 or 1 entries to indicate
    set membership based on that metadata column.
    '''
    #read the table into node data and build lists of column entries
    with open(in_file,'rU') as f:
        reader = csv.DictReader(f, delimiter='\t')
        nodes = []
        column_data = {}
        for fn in reader.fieldnames: #@UnusedVariable
            column_data.update({fn:[]})
        for line in reader:
            tmp_dict = {}
            tmp_column_list = []
            for k in line.keys():
                column_data[k].append(line[k])
                if k == 'name':
                    tmp_dict.update({k:line[k]})
                else:
                    if line[k] == '1':
                        tmp_column_list.append(k)
            tmp_dict.update({'networks':tmp_column_list})
            nodes.append(tmp_dict)
    
    print(column_data)
    #write the nodes to file
    with open(nodes_file,'w') as f:
        f.write(json.dumps(nodes))



[docs]def build_nodes_from_text(in_file,out_file):
    '''
    build json array of nodes suitable for use in d3.js
    '''
    #get the number of lines in the input file
    num_lines = cjf_text.line_count(in_file)
    
    #read the input file
    with open(in_file,'rU') as f:
        lines = f.readlines()
        
    #build the json array
    json_array = []
    for ii,line in enumerate(lines):
        json_array.append({"name":line.strip(),"group":1, "rank":(float(num_lines) - ii)/num_lines})
    
    #write to file
    json_text = json.dumps(json_array)
    with open(out_file,'w') as f:
        f.write(json_text)



[docs]def build_links_from_text(in_file,out_file,link_color='red',exclude_annot=''):
    '''
    builds a json array of links suitable for use in d3.js
    '''
    #read the input file
    with open(in_file,'rU') as f:
        lines = f.readlines()
    
    #build lists of entries per line
    line_lists = []
    for line in lines:
        line_text = line.strip()
        line_lists.append(line_text.split(','))
    
    #build links
    link_store = []
    json_array = []
    for ii,line_list in enumerate(line_lists):
        link_inds = []
        for item in line_list:
            if exclude_annot not in item:
                inds = get_line_links(item,line_lists)
                link_inds.extend(inds)
        for link_ind in link_inds:
            if not ii == link_ind:
                if not (link_ind,ii) in link_store:#make sure we don't already have the link
                    link_store.append((ii,link_ind))
                    json_array.append({"source":ii,"target":link_ind,"value":40,"color":link_color})
    json_text = json.dumps(json_array)
    
    #write to file
    with open(out_file,'w') as f:
        f.write(json_text)
            


[docs]def get_line_links(query,line_lists):
    '''
    find the lines that contain matches for the given query
    '''
    link_inds = []
    for ii,line_list in enumerate(line_lists):
        for item in line_list:
            if not item == '':
                if item == query:
                    link_inds.append(ii)
    
    return link_inds
