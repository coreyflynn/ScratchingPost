    def __init__(self,queries=None,database=None,database_cid=None,database_rid=None,

        '''
        Constructor
        '''
        self.queries = queries
        self.results = []
        self.database = database
        self.database_cid = database_cid
        self.database_rid = database_rid
        self.database_meta = database_meta
        self.database_src = ''
    
    def __repr__(self):
        return ('Query(queries=%r'
                ',database=%r'
                ',database_cid=%r'
                ',database_rid=%r'
                ',database_meta=%r)' % (self.queries,self.database,
                                       self.database_cid, self.database_rid,
                                       self.database_meta)) 
    def __str__(self):
        return '\n'.join(['queries: ' + str(type(self.queries)),
                'database: '  + str(type(self.database)),
                'database_cid: ' + str(type(self.database_cid)),
                'database_rid: ' + str(type(self.database_rid)),
                'database_meta: ' + str(type(self.database_meta))]) 
    
    
    def _add_to_database(self,database,database_cid,database_rid,database_meta=None):
        '''
        adds the numpy array to the database along with the given cids
        '''
        #check to see if the database already has content.  If it does, make sure that we
        #can add to it properly and join them. Otherwise just pass the given arguments to 
        #the object
        if self.database is not None:
            if self.database.shape[0] != database.shape[0]:
                raise QueryException('database shapes must match')
            self.database = numpy.concatenate((self.database,database),axis=1)
        else:
            self.database = database
            
        #check for existing cid data and make sure new cid data is in a list
        if type(database_cid) is not list:
            raise QueryException('database_cid must be a list')
        if self.database_cid is not None:
            self.database_cid.extend(database_cid)
        else:
            self.database_cid = database_cid
        
        #check for existing rid data and make sure new rid data is in a list    
        if type(database_rid) is not list:
            raise QueryException('database_rid must be a list')
        if self.database_rid is not None:
            self.database_rid.extend(database_rid)
        else:
            self.database_rid = database_rid
            
        #update the database meta data
        if database_meta is not None:
            self.database_meta.update(database_meta)

        '''

        self._add_to_database(database, database_cid, database_rid,
                              database_meta = database_meta)
    


[docs]    def spearman_query(self,query=None,cid=None,database=None):
        '''
        query the database with all of the deltas in self.queries or in the specified query
        matrix 
        '''
        #instantiate a progressbar object
        progress_bar = progress.DeterminateProgressBar('Spearman Query')
        
        #temporarily ignore divide by zero errors
        numpy.seterr(divide='ignore')
        
        #run the query
        if query is None:
            query = self.queries
        if database is None:
            database = self.database
        if query is not None:
            query_results = numpy.zeros([self.database.shape[1],query.shape[1]])
            if self.database.shape[0] != query.shape[0]:
                raise QueryException('database and query shapes must match')
            total_query_num = query.shape[1]*self.database.shape[1]
            if cid is None:
                cid = range(query.shape[1])
            query_sort = query.argsort(axis=0).argsort(axis=0)
            database_sort = database.argsort(axis=0).argsort(axis=0)
            for i in range(query.shape[1]):
                for j in range(database.shape[1]):
                    progress_bar.update('running query %s' %(cid[i],), i*self.database.shape[1]+j, total_query_num)
                    query_results[j,i] = scipy.stats.pearsonr(query_sort[:,i],database_sort[:,j])[0]
            numpy.seterr(divide=None)
            progress_bar.clear()
            self.results.append(QueryResult(query_type = 'spearman_corr',
                                            data = query_results,
                                            cid = cid,
                                            database_id = self.database_cid))
        
        else:
            numpy.seterr(divide=None)
            raise QueryException('query is None, nothing to search!')
        


[docs]    def rrho_query(self,query=None,cid=None,database=None,start=50,step=100):
        '''
        query the database with all of the deltas in self.queries or in the specified query
        matrix using rank-rank hypergeometric overlap 
        '''
        #instantiate a progressbar object
        progress_bar = progress.DeterminateProgressBar('RRHO Query')
        
        #run the query
        if query is None:
            query = self.queries
        if database is None:
            database = self.database
        if query is not None:
            query_results = numpy.zeros([self.database.shape[1],query.shape[1]])
            query_len = query.shape[0]
            database_len = self.database.shape[0]
            if database_len != query_len:
                raise QueryException('database and query shapes must match')
            total_query_num = query.shape[1]*self.database.shape[1]
            if cid is None:
                cid = range(query.shape[1])
            query_sort = query.argsort(axis=0).argsort(axis=0)
            database_sort = database.argsort(axis=0).argsort(axis=0)
            for i in range(query.shape[1]):
                for j in range(database.shape[1]):
                    progress_bar.update('running query %s' %(cid[i],), i*self.database.shape[1]+j, total_query_num)
                    rrho_list = []
                    for ii in range(start,query_len,step):
                        for jj in range(start,database_len,step):
                            set1 = set(query_sort[0:ii,i])
                            set2 = set(database_sort[0:jj,j])
                            overlap = len(set1.intersection(set2))
                            rrho = scipy.stats.hypergeom.cdf(overlap,query_len,ii,jj)
                            #normalize to expected and log transform
                            expected_overlap = float(ii)*(jj/query_len)
                            if overlap > expected_overlap:
                                rrho = 1-rrho
                                if rrho == 0:
                                    rrho = 1
                                rrho_log = +abs(math.log10(rrho))
                            else:
                                rrho_log = -abs(math.log10(rrho))
                            
                            rrho_list.append(rrho_log)
                    query_results[j,i] = numpy.max(rrho_list)
            progress_bar.clear()
            self.results.append(QueryResult(query_type = 'RRHO',
                                            data = query_results,
                                            cid = cid,
                                            database_id = self.database_cid))
        else:
            raise QueryException('query is None, nothing to search!')



[docs]    def gsea_query(self,query=None,cid=None,database=None,gene_set_size=50):
        '''
        query the database with all of the deltas in self.queries or in the specified query
        matrix using preranked gsea
        '''
        #instantiate a progressbar object
        progress_bar = progress.DeterminateProgressBar('GSEA Query')
        
        #run the query
        if query is None:
            query = self.queries
        if database is None:
            database = self.database
        if query is not None:
            query_results = numpy.zeros([self.database.shape[1],query.shape[1]])
            query_len = query.shape[0]
            database_len = self.database.shape[0]
            if database_len != query_len:
                raise QueryException('database and query shapes must match')
            total_query_num = query.shape[1]*self.database.shape[1]
            if cid is None:
                cid = range(query.shape[1])
            query_sort = query.argsort(axis=0)
            database_sort = database.argsort(axis=0)
            for i in range(query.shape[1]):
                for j in range(database.shape[1]):
                    progress_bar.update('running query %s' %(cid[i],), i*self.database.shape[1]+j, total_query_num)
                    gene_set = numpy.take(self.database_rid,query_sort[:,i])[0:gene_set_size]
                    ranked_list = numpy.take(self.database_rid,database_sort[:,j])
                    ES = gsea.preranked(gene_set, ranked_list)
                    query_results[j,i] = ES
                    
            progress_bar.clear()
            self.results.append(QueryResult(query_type = 'GSEA',
                                            data = query_results,
                                            cid = cid,
                                            database_id = self.database_cid))
        else:
            raise QueryException('query is None, nothing to search!')
        


[docs]    def show_break_out_table(self,query_inds,query_type='spearman',labels=None,scale=1):
        '''
        runs queries from the given indices into the dataset and displays them as a break
        out result table 
        '''
        #run the query
        self.spearman_query(query=self.database[:,query_inds],cid=labels)
        
        #if labels are not specified, get the labels out of the database_cid
        if labels is None:
            labels = query_inds
        
        #get the table values
        query_result = self.results[-1]
        table_vals = [ ['' for c in query_inds] for r in query_inds] #@UnusedVariable
        ranks = query_result.data.argsort(axis=0).argsort(axis=0)
        num_ranks = ranks.shape[0]
        for ii in range(len(query_inds)):
            for jj,ind in enumerate(query_inds):
                table_vals[ii][jj] = num_ranks-ranks[ind,ii]
        
        colors = [ [(1,0,0) for c in query_inds] for r in query_inds] #@UnusedVariable
        for ii in range(len(query_inds)):
            for jj in range(len(query_inds)):
                if table_vals[ii][jj] < num_ranks*0.05:
                    colors[ii][jj] = (0,1,0)
        
        #show the table
        the_table = plt.table(cellText=table_vals,
                  colWidths = [0.1]*len(query_inds),
                  rowLabels=labels,
                  colLabels=labels,
                  cellColours=colors,
                  rowLoc='center',
                  loc='center')
        the_table.scale(scale,scale)
        the_table.auto_set_font_size(False)
        
        plt.axis('off')
        plt.show()

[docs]class QueryResult:
    '''
    class to hold the results of queries inside of a top level Query object
    '''
    def __init__(self, query_type='', data=[], cid=[], database_id=[]):
        '''
        Constructor
        '''
        self.query_type= query_type
        self.data = data
        self.cid = cid
        self.database_cid = database_id
        self.rank_map = {}
        
        self._make_rank_map()
        
    def __repr__(self):
        return ('QueryResult(query_type=%r'
                ',data=%r'
                ',cid=%r'
                ',database_cid=%r'
                ',rank_map=%r)' % (self.query_type,self.data,
                                       self.cid, self.database_cid,
                                       self.rank_map))
    def __str__(self):
        return '\n'.join(['query_type: %s' %(self.query_type,),
                          'number of queries: %s' %(self.data.shape[1],),
                          'size of database: %s' %(self.data.shape[0],)]) 
    
    def _make_rank_map(self):
        '''
        makes the mapping of cids to dictionaries of (database_cid,rank) tuple lists 
        '''
        for i,cid in enumerate(self.cid):
            inds = numpy.argsort(self.data[:,i])
            sorted_database_id = numpy.take(self.database_cid,inds).tolist()
            sorted_database_id.reverse()
            tmp_tuple = zip(sorted_database_id,range(len(sorted_database_id)))
            self.rank_map.update({cid:tmp_tuple})

[docs]    def write_table(self,path):
        '''
        writes the query results to a table at the given path
        '''
        with open(path,'w') as f:
            for cid in self.cid:
                f.write('\t' + str(cid))
            f.write('\n')
            for ii in range(len(self.database_cid)):
                f.write(str(self.database_cid[ii]))
                for jj in range(len(self.cid)):
                    f.write('\t' + str(self.data[ii,jj]))
                f.write('\n')
