from SOAPpy import WSDL

import cjf.cmap.util.progress as update


[docs]class KEGGServer(object):
    '''
    class to handle easy interaction with the kegg server api using SOAPpy
    '''

    def __init__(self):
        '''
        Constructor
        '''
        self.server = WSDL.Proxy('http://soap.genome.jp/KEGG.wsdl')
    

[docs]    def get_pathways_from_symbol(self,symbol):
        '''
        returns a string containing the KEGG pathway annotations to which the given symbol
        belongs 
        '''
        pathways_array = []
        kegg_id = self.server.btit('hsa:' + symbol)
        kegg_pathways = self.server.get_pathways_by_genes([kegg_id.split(' ')[0]])
        for path in kegg_pathways:
            pathways_array.append(self.server.btit(path))
        return pathways_array
 


[docs]def annotate_grp(grp_path,out_path):
    '''
    attach KEGG pathway annotations to a list of gene symbols provided as a grp file.  A
    table of symbols and associated annotations is written to out_path
    '''
    #open the input file and extract the symbols from the file
    with open(grp_path,'rU') as f:
        symbols = f.readlines()
    
    #start a KEGGServer for later use
    server = KEGGServer()
    
    #start a update indicator
    progress_bar = update.DeterminateProgressBar('KEGGServer')
    
    #for each provided symbol, look up the KEGG pathway annotation(s)
    with open(out_path,'w') as f:
        for ii,symbol in enumerate(symbols):
            f.write(symbol.strip() + '\t')
            progress_bar.update('annotating ' + symbol.strip(), ii, len(symbols))
            pathways = server.get_pathways_from_symbol(symbol.strip())
            if len(pathways) > 0: 
                pathway_ids = []
                pathway_texts = []
                for jj, pathway in enumerate(pathways):
                    pathway_id = pathway.partition(' ')[0]
                    pathway_text = pathway.partition(' ')[2].split(' - ')[0]
                    pathway_ids.append(pathway_id)
                    pathway_texts.append(pathway_text)
                for kk,pid in enumerate(pathway_ids):
                    if kk == 0:
                        f.write(pid)
                    else:
                        f.write(',' + pid)
                f.write('\t')
                for kk,text in enumerate(pathway_texts):
                    if kk == 0:
                        f.write(text)
                    else:
                        f.write(',' + text)
                f.write('\n')
            else:
                f.write('\t\n')
