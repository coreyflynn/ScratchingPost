[docs]    def add_edges_from_query_result(self,QueryResult,thresh=0.01):
        '''
        adds edges to the graph based on the results of the a query result set.  results in
        the top thresh % of the data are included
        '''
        #instantiate a progressbar object
        progress_bar = progress.DeterminateProgressBar('Adding Edges')
        
        #find the number of results in the query set and determine how many to use
        
        num_results = len(QueryResult.rank_map[QueryResult.rank_map.iterkeys().next()])
        num_sig_results =  int(num_results*thresh)
        
        #for each query in the query result set, find the id of the top thresh % of results
        node_id_dict = nx.get_node_attributes(self, 'id')
        inds,ids = zip(*node_id_dict.items())
        num_queries = len(QueryResult.rank_map.keys())
        it = 0
        for k in QueryResult.rank_map.keys():
            progress_bar.update('processing query %s' %(k,), it, num_queries)
            it += 1
            sig_results = QueryResult.rank_map[k][0:num_sig_results]
            sig_result_edges = list(itertools.combinations(sig_results,2))
            sig_result_id1 = [x[0][0] for x in sig_result_edges]
            sig_result_id2 = [x[1][0] for x in sig_result_edges]
            sig_result_node1 = [ids.index(x) for x in sig_result_id1]
            sig_result_node2 = [ids.index(x) for x in sig_result_id2]
            sig_result_weight = [num_sig_results - (float(x[0][1]+x[1][1]))/2 for x in sig_result_edges]
            weighted_result_edges = zip(sig_result_node1,sig_result_node2,sig_result_weight)
            self.add_weighted_edges_from(weighted_result_edges)
        
        progress_bar.clear()
        

    


[docs]    def show_connected_graph(self):
        '''
        draws the subgraph of connected components to the screen
        '''
        #get the connected components of the graph and place all of their indices into one
        #list if they are larger than one compoent
        cc = nx.connected_components(self)
        all_large_cc = []
        for c in cc:
            if len(c) > 1:
                all_large_cc.extend(c)
        
        #make a subgraph and display it
        subG = self.subgraph(all_large_cc)
        nx.draw_spring(subG, alpha=0.5, style='dotted', with_labels=False)
        plt.show()
    


[docs]    def walk_connections(self,runs=1000,steps=100, return_dict=False):
        '''
        walks the connections of the graph and keeps track of the proportion of the steps
        spent on each node touched.  At each step, one edge is selected at random 
        from the current node and the edge is traversed
        '''
        #instantiate a progressbar object
        progress_bar = progress.DeterminateProgressBar('Walking Connections')
        
        #get connected components
        cc = nx.connected_components(self)
        connected_nodes = []
        for c in cc:
            if len(c) > 1:
                connected_nodes.extend(c)
        
        
        node_counts = {}
        node_dist = {}
        run_num = 0
        while run_num < runs:
            progress_bar.update('run %s of %s' %(run_num,runs), run_num, runs)
            current_node = random.choice(connected_nodes)
            step = 0;
            while step < steps:
                #log the presence of the walker at current_node
                if current_node not in node_counts.keys():
                    node_counts[current_node] = 1
                else:
                    node_counts[current_node] = node_counts[current_node] + 1 
                
                #choose and edge and traverse it
                if current_node not in node_dist.keys():
                    current_edges = [k for k in self[current_node].keys()]
                    current_edge_weights = []
                    current_edge_dist = []
                    for e in current_edges:
                        for key in self[current_node][e].keys():
                            tmp_weights = []
                            tmp_weights.append(self[current_node][e][key]['weight'])
                        tmp_weight_mean = np.mean(tmp_weights)
                        current_edge_weights.append(tmp_weight_mean)
                    
                    for i in range(len(current_edges)):
                        current_edge_dist.extend([current_edges[i]]*int(round(current_edge_weights[i])))
                    node_dist[current_node] = current_edge_dist
                    
                current_node = random.choice(node_dist[current_node])
                
                #update the step counter
                step += 1
            run_num += 1
        progress_bar.clear()
        if return_dict:
            return node_counts
        else:
            node_tuple_list = [(k,v) for k,v in node_counts.items()]
            node_tuple_list.sort(key=lambda node_tuple_list: node_tuple_list[1], reverse=True)
            return [x[0] for x in node_tuple_list]
        


[docs]    def get_pagerank(self, return_dict=False):
        '''
        gets the page rank of all nodes in the connected components of the graph and returns
        them as a dictionary
        '''
        #get connected components
        cc = nx.connected_components(self)
        connected_nodes = []
        for c in cc:
            if len(c) > 1:
                connected_nodes.extend(c)
        
        #build a sub-graph containing only the connected components.  This graph must be a
        #non-MultiGraph for pagerank to work
        G = nx.Graph()
        G.add_nodes_from(connected_nodes)
        node1 = []
        node2 = []
        weight = []
        for node in connected_nodes:
            edges = [k for k in self[node].keys()]
            for e in edges:
                for key in self[node][e].keys():
                    tmp_weights = []
                    tmp_weights.append(self[node][e][key]['weight'])
                tmp_weight_mean = np.mean(tmp_weights)
                node1.append(node)
                node2.append(e)
                weight.append(tmp_weight_mean)
        G.add_weighted_edges_from(zip(node1,node2,weight))
        
        #get the pagerank of the graph and return it as a dictionary
        pr = nx.pagerank(G)
        if return_dict:
            return pr
        else:
            pr_tuple_list = [(k,v) for k,v in pr.items()]
            pr_tuple_list.sort(key=lambda pr_tuple_list: pr_tuple_list[1], reverse=True)
            return [x[0] for x in pr_tuple_list]
