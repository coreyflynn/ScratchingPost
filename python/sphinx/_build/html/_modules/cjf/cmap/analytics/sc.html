        '''
        write only the pert_id portion of the the ids of the points in the quad to file.  If
        unique=True, no duplicate pert_ids are written to file
        '''
        written = []
        with open(out_path,'w') as f:
            for ids in self.ids:
                pert = ids.split(':')[2]
                if unique:
                    if pert not in written:
                        written.append(pert)
                        f.write(pert + '\n')
                else:
                    f.write(pert + '\n')

[docs]    def update_quad_info(self,clear=True):
        '''
        take the current value of s and c and compute the quadrant membership of all points
        based on cutoffs
        '''
        #clear quads if needed
        if clear:
            self.top_right.clear_points()
            self.top_left.clear_points()
            self.bottom_left.clear_points()
            self.bottom_right.clear_points()
        
        #zip s and c for convenient iteration
        sc = zip(self.s,self.c)
        
        #top right
        indices  = [i for i,v in enumerate(sc) if (v[0] >= self.s_thresh) & (v[1] >= self.c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.top_right.add_points(ids, indices)
        
        #top left
        indices  = [i for i,v in enumerate(sc) if (v[0] >= self.s_thresh) & (v[1] < self.c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.top_left.add_points(ids, indices)
        
        #bottom right
        indices  = [i for i,v in enumerate(sc) if (v[0] < self.s_thresh) & (v[1] >= self.c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.bottom_right.add_points(ids, indices)
        
        #bottom left
        indices  = [i for i,v in enumerate(sc) if (v[0] < self.s_thresh) & (v[1] < self.c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.bottom_left.add_points(ids, indices)



[docs]    def add_sc_from_sigstat(self,sigstat_path,cc_field='cc_q75',ss_field='sig_strength',id_field='id'):
        '''
        reads the input sigstats file and adds the values for the SS and CC according to the
        parameters specified
        '''
        #ensure that s and c are lists not numpy arrays
        self.c = list(self.c)
        self.s = list(self.s)
        
        #set up a dictionary reader to parse sigstats
        f = open(sigstat_path,'rU')
        reader = csv.DictReader(f, delimiter='\t')
        
        #read in the specified cc value and the signature strength
        for row in reader:
            try:
                self.c.append(float(row[cc_field]))
                self.s.append(float(row[ss_field]))
                self.pid.append(row[id_field])
            except KeyError as E:
                print ('omitting %s:' %(sigstat_path,))
                print ('could not find column "%s"' % (E.args[0]))
                return
        
        #add the file path to self.src
        self.src.append(sigstat_path)
        


[docs]    def add_sc_from_gct(self,src,ss_column_name='distil_ss', cc_column_name='distil_cc_q75'):
        '''
        reads the meta data of the given gct or gctx file and adds ss and cc values for
        samples in the file
        '''
        #read in the gct data
        gct_obj = gct.GCT(src=src)
        
        #grab the pid, ss, and cc data as well as ss and cc cutoffs
        s = gct_obj.get_column_meta(ss_column_name)
        c = gct_obj.get_column_meta(cc_column_name)
        pid = gct_obj.get_column_meta('id')
        s_thresh = float(gct_obj.get_column_meta('distil_ss_cutoff')[0])
        c_thresh = float(gct_obj.get_column_meta('distil_cc_cutoff')[0])
        
        #ensure that s and c are lists not numpy arrays
        self.c = list(self.c)
        self.s = list(self.s)
        
        #convert ss and cc into float values
        s = [float(x) for x in s]
        c = [float(x) for x in c]
        
        #add pid, ss, and cc to the existing data
        self.pid.extend(pid)
        self.s.extend(s)
        self.c.extend(c)
        
        #set new thresholds
        self.s_thresh = s_thresh
        self.c_thresh = c_thresh

[docs]    def filter_sc(self,include=None,exclude=None,control=None, destructive=True):
        '''
        filters the c and s arrays according to the include,exclude, and control parameters.
        If control is set, self.control_inds is set to the correct value
        '''
        #convert the s and c lists to numpy arrays
        self.c = numpy.array(self.c)
        self.s = numpy.array(self.s)
        
        #set up internal variables if destructive is false
        if not destructive:
            c = self.c
            s = self.s
            pid = self.pid
        
        #include only those entries specified and exclude those entries specified
        if exclude:
            exclude_inds = []
            if isinstance(exclude,basestring):
                exclude = [exclude]
            for entry in exclude:
                pattern = re.compile(entry,re.IGNORECASE)
                entry_inds = [i for i,v in enumerate(self.pid) if pattern.search(v)]
                for eind in entry_inds:
                    exclude_inds.append(eind)
            full_inds_set = set(range(len(self.s)))
            exclude_removed_set = full_inds_set.difference(set(exclude_inds))
            if destructive:
                self.c = self.c[list(exclude_removed_set)]
                self.s = self.s[list(exclude_removed_set)]
                self.pid = [self.pid[i] for i in exclude_removed_set]
            else:
                c = self.c[list(exclude_removed_set)]
                s = self.s[list(exclude_removed_set)]
                pid = [self.pid[i] for i in exclude_removed_set]
        
        if include:
            include_inds = []
            if isinstance(include,basestring):
                include = [include]
            for entry in include:
                pattern = re.compile(entry,re.IGNORECASE)
                entry_inds = [i for i,v in enumerate(self.pid) if pattern.search(v)]
                for eind in entry_inds:
                    include_inds.append(eind)
            if destructive:
                self.c = self.c[include_inds]
                self.s = self.s[include_inds]
                self.pid = [self.pid[i] for i in include_inds]
            else:
                c = self.c[include_inds]
                s = self.s[include_inds]
                pid = [self.pid[i] for i in include_inds]
                
                
        #pull out the control inds if they are specified
        if control:
            if destructive:
                self.control_inds = []
            else:
                control_inds = []
            if isinstance(control,basestring):
                control = [control]
            for entry in control:
                pattern = re.compile(entry,re.IGNORECASE)
                if destructive:
                    entry_inds = [i for i,v in enumerate(self.pid) if pattern.search(v)]
                else:
                    entry_inds = [i for i,v in enumerate(pid) if pattern.search(v)]
                for eind in entry_inds:
                    if destructive:
                        self.control_inds.append(eind)
                    else:
                        control_inds.append(eind)
        else:
            if not destructive:
                control_inds = None
        
        #return s,c,pid,and control_inds if destructive is false
        if not destructive:
            return (s,c,pid,control_inds)
        #convert the numpy arrays back to lists
        self.c = self.c.tolist()
        self.s = self.s.tolist()
        
        #update the quadrant information
        self.update_quad_info()
    


[docs]    def plot(self,out=None,xlabel='CC',ylabel='SS'
                     ,title=None, control=['DMSO']
                     ,include=None, exclude=None
                     ,pos_con=['SAHA','Trichostatin A','Fulvestrant',
                               'Genistein','LY-294002','Sirolimus',
                               'Thioridazine','Withaferin A',
                               'shERG#10']):
        '''
        plots and sc_plot.  The data is assumed to fall between a signature strength of 
        [0 20] and correlations between [-1 1]. Unless out is specified,the figure is drawn 
        on screen
        '''
        #include controls if needed
        (s,c,pid,control_inds) = self.filter_sc(control=control, include=include, exclude=exclude, destructive=False) #@UnusedVariable
        
        #include pos_cons if needed
        (s,c,pid,pos_con_inds) = self.filter_sc(control=pos_con, include=include, exclude=exclude, destructive=False) #@UnusedVariable
        
        
        #convert the s and c lists to numpy arrays
        c = numpy.array(c)
        s = numpy.array(s)
        
        #build the figure container and axis
        fig = plt.figure()
        ax = fig.add_subplot(111)

        #grab controls for later plotting
        if control_inds or pos_con_inds:
            if not control_inds:
                control_inds = []
            if not pos_con_inds:
                pos_con_inds = []
            control_c = c[control_inds]
            control_s = s[control_inds]
            pos_con_c = c[pos_con_inds]
            pos_con_s = s[pos_con_inds]
            control_plus_pos_con_inds = []
            control_plus_pos_con_inds.extend(pos_con_inds)
            exp_inds = list(set(range(len(c))).difference(set(control_plus_pos_con_inds)))
            exp_c = c[exp_inds]
            exp_s = s[exp_inds]
        else:
            exp_c = c
            exp_s = s
        
        #find the number of data points in each quadrant
        top_right_s = exp_s[(exp_s >= self.s_thresh) & (exp_c >= self.c_thresh)] 
        top_left_s = exp_s[(exp_s >= self.s_thresh) & (exp_c < self.c_thresh)]
        bot_right_s = exp_s[(exp_s < self.s_thresh) & (exp_c >= self.c_thresh)]
        bot_left_s = exp_s[(exp_s < self.s_thresh) & (exp_c < self.c_thresh)]
        
        #build quadrant colors
        top_right_color = (1-top_right_s.size/float(exp_s.size),
                           1,
                           1-top_right_s.size/float(exp_s.size))
        top_left_color = (1-top_left_s.size/float(exp_s.size),
                           1,
                           1-top_left_s.size/float(exp_s.size))
        bot_right_color = (1-bot_right_s.size/float(exp_s.size),
                           1,
                           1-bot_right_s.size/float(exp_s.size))
        bot_left_color = (1-bot_left_s.size/float(exp_s.size),
                           1,
                           1-bot_left_s.size/float(exp_s.size))
        
        #build quadrant rectangles
        top_right_rect = patches.Rectangle((self.c_thresh,self.s_thresh),1-self.c_thresh,20-self.s_thresh, 
                                           color = top_right_color,
                                           zorder = 1)
        ax.add_patch(top_right_rect)
        top_left_rect = patches.Rectangle((-1,self.s_thresh),1+self.c_thresh,20-self.s_thresh, 
                                           color = top_left_color, 
                                           zorder = 1)
        ax.add_patch(top_left_rect)
        bot_right_rect = patches.Rectangle((self.c_thresh,0),1-self.c_thresh,self.s_thresh, 
                                           color = bot_right_color, 
                                           zorder = 1)
        ax.add_patch(bot_right_rect)
        bot_left_rect = patches.Rectangle((-1,0),1+self.c_thresh,self.s_thresh, 
                                           color = bot_left_color, 
                                           zorder = 1)
        ax.add_patch(bot_left_rect)
        
        #build the scatter plot and label the axis
        ax.scatter(exp_c,exp_s, zorder = 2, label='pert')
        if control_inds:
            ax.scatter(control_c,control_s,facecolor='r',zorder=3, label='control')
        if pos_con_inds:
            ax.scatter(pos_con_c,pos_con_s,facecolor='burlywood',zorder=4, label='pos con')
        if control_inds or pos_con_inds:
            plt.legend(loc='center left')
        ax.set_xlim(-1,1)
        ax.set_ylim(0,20)
        
        
        #add text annotations
        props = dict(boxstyle='round', facecolor='wheat')
        ax.text(0.05, 0.90, '%.1f%%' %(top_left_s.size/float(exp_s.size)*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        ax.text(0.8, 0.90, '%.1f%%' %(float(top_right_s.size)/exp_s.size*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        ax.text(0.05, 0.05, '%.1f%%' %(bot_left_s.size/float(exp_s.size)*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        ax.text(0.8, 0.05, '%.1f%%' %(bot_right_s.size/float(exp_s.size)*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        
        #add cutoff annotation
        ax.axhline(y=self.s_thresh,linewidth=2,color='k')
        ax.axvline(x=self.c_thresh,linewidth=2,color='k')
        
        #add labels, add title, and increase tick font size
        plt.xlabel(xlabel, fontsize=20)
        plt.ylabel(ylabel, fontsize=20)
        if title:
            plt.title(title, fontsize=20)
        for tick in ax.xaxis.get_major_ticks():
            tick.label1.set_fontsize(20)
        for tick in ax.yaxis.get_major_ticks():
            tick.label1.set_fontsize(20)
        
        #show on screen unless out is specified.  If it is, write the figure to the path in out
        if not out:
            plt.show()
        else:
            plt.savefig(out)

[docs]    def update_annotation_dict_from_table(self,annotation_table):
        '''
        updates the annotation dictionary from a table containing the ids under a column 
        headed by 'name' and any number of additional columns with headers 
        '''
        #ensure that we start with an empty annotation dictionary
        self.annot_dict = {}
        
        #read the table into a dictionary containing the headers as keys and a list of the
        #rows containing 1's for each column as the values
        with open(annotation_table,'rU') as f:
            reader = csv.DictReader(f, delimiter='\t')
            for fn in reader.fieldnames:
                if 'name' not in fn:
                    self.annot_dict.update({fn:[]})
            for line in reader:
                for k in line.keys():
                    if 'name' not in  k:
                        if line[k] == '1':
                            self.annot_dict[k].append(line['name'])
