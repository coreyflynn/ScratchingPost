[docs]    def update_quad_info(self,s_thresh = 6, c_thresh = 0.25):
        '''
        take the current value of s and c and compute the quadrant membership of all points
        based on cutoffs
        '''
        #zip s and c for convenient iteration
        sc = zip(self.s,self.c)
        
        #top right
        indices  = [i for i,v in enumerate(sc) if (v[0] >= s_thresh) & (v[1] >= c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.top_right.add_points(ids, indices)
        
        #top left
        indices  = [i for i,v in enumerate(sc) if (v[0] >= s_thresh) & (v[1] < c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.top_left.add_points(ids, indices)
        
        #bottom right
        indices  = [i for i,v in enumerate(sc) if (v[0] < s_thresh) & (v[1] >= c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.bottom_right.add_points(ids, indices)
        
        #bottom left
        indices  = [i for i,v in enumerate(sc) if (v[0] < s_thresh) & (v[1] < c_thresh)]
        ids = [self.pid[i] for i in indices]
        self.bottom_left.add_points(ids, indices)



[docs]    def add_sc_from_sigstat(self,sigstat_path,cc_field='cc_mean',ss_field='sig_strength',id_field='id'):
        '''
        reads the input sigstats file and adds the values for the SS and CC according to the
        parameters specified
        '''
        #ensure that s and c are lists not numpy arrays
        self.c = list(self.c)
        self.s = list(self.s)
        
        #set up a dictionary reader to parse sigstats
        f = open(sigstat_path,'rU')
        reader = csv.DictReader(f, delimiter='\t')
        
        #read in the specified cc value and the signature strength
        for row in reader:
            try:
                self.c.append(float(row[cc_field]))
                self.s.append(float(row[ss_field]))
                self.pid.append(row[id_field])
            except KeyError as E:
                print ('omitting %s:' %(sigstat_path,))
                print ('could not find column "%s"' % (E.args[0]))
                return
        
        #add the file path to self.src
        self.src.append(sigstat_path)

        #get the sigstats path from the specified brew folder

[docs]    def add_sc_from_grp(self,grp_path):
        '''
        reads the input sigstats file from all brew folders given in the specified grp file.
        The values for the SS and CC in the file are added. The grp file should contain one 
        brew folder per line
        '''
        #read in all of the lines of the grp file
        brew_names = []
        with open(grp_path,'rU') as f:
            for line in f:
                brew_names.append(line.strip())
        
        #get the sigstats path from each specified brew folder and add the data
        for brew_name in brew_names:
            sigstat_path = os.path.join('/xchip/cogs/data/brew',brew_name,
                                        'sigstat',brew_name + '_SIGSTATS.txt')
            self.add_sc_from_sigstat(sigstat_path)
        
        #update the quadrant information 
        self.update_quad_info()



[docs]    def filter_sc(self,include=None,exclude=None,control=None, destructive=True):
        '''
        filters the c and s arrays according to the include,exclude, and control parameters.
        If control is set, self.control_inds is set to the correct value
        '''
        #convert the s and c lists to numpy arrays
        self.c = numpy.array(self.c)
        self.s = numpy.array(self.s)
        
        #set up internal variables if destructive is false
        if not destructive:
            c = self.c
            s = self.s
            pid = self.pid
        
        #include only those entries specified and exclude those entries specified
        if exclude:
            exclude_inds = []
            if isinstance(exclude,basestring):
                exclude = [exclude]
            for entry in exclude:
                pattern = re.compile(entry,re.IGNORECASE)
                entry_inds = [i for i,v in enumerate(self.pid) if pattern.search(v)]
                for eind in entry_inds:
                    exclude_inds.append(eind)
            full_inds_set = set(range(len(self.s)))
            exclude_removed_set = full_inds_set.difference(set(exclude_inds))
            if destructive:
                self.c = self.c[list(exclude_removed_set)]
                self.s = self.s[list(exclude_removed_set)]
                self.pid = [self.pid[i] for i in exclude_removed_set]
            else:
                c = self.c[list(exclude_removed_set)]
                s = self.s[list(exclude_removed_set)]
                pid = [self.pid[i] for i in exclude_removed_set]
        
        if include:
            include_inds = []
            if isinstance(include,basestring):
                include = [include]
            for entry in include:
                pattern = re.compile(entry,re.IGNORECASE)
                entry_inds = [i for i,v in enumerate(self.pid) if pattern.search(v)]
                for eind in entry_inds:
                    include_inds.append(eind)
            if destructive:
                self.c = self.c[include_inds]
                self.s = self.s[include_inds]
                self.pid = [self.pid[i] for i in include_inds]
            else:
                c = self.c[include_inds]
                s = self.s[include_inds]
                pid = [self.pid[i] for i in include_inds]
                
                
        #pull out the control inds if they are specified
        if control:
            if destructive:
                self.control_inds = []
            else:
                control_inds = []
            if isinstance(control,basestring):
                control = [control]
            for entry in control:
                pattern = re.compile(entry,re.IGNORECASE)
                if destructive:
                    entry_inds = [i for i,v in enumerate(self.pid) if pattern.search(v)]
                else:
                    entry_inds = [i for i,v in enumerate(pid) if pattern.search(v)]
                for eind in entry_inds:
                    if destructive:
                        self.control_inds.append(eind)
                    else:
                        control_inds.append(eind)
        else:
            if not destructive:
                control_inds = None
        
        #return s,c,pid,and control_inds if destructive is false
        if not destructive:
            return (s,c,pid,control_inds)
        #convert the numpy arrays back to lists
        self.c = self.c.tolist()
        self.s = self.s.tolist()
    


[docs]    def plot(self,s_thresh=6,c_tresh=0.25,out=None,xlabel='CC',ylabel='SS'
                     ,title=None, control=None, include=None, exclude=None):
        '''
        plots and sc_plot.  The data is assumed to fall between a signature strengths of 
        [0 20] and correlations between [-1 1]. Unless out is specified,the figure is drawn 
        on screen
        '''
        #include controls if needed
        (s,c,pid,control_inds) = self.filter_sc(control=control, include=include, exclude=exclude, destructive=False)
        
        #convert the s and c lists to numpy arrays
        c = numpy.array(c)
        s = numpy.array(s)
        
        #build the figure container and axis
        fig = plt.figure()
        ax = fig.add_subplot(111)
        
        #grab controls for later plotting
        if control_inds:
            control_c = c[control_inds]
            control_s = s[control_inds]
            exp_inds = list(set(range(len(c))).difference(set(control_inds)))
            exp_c = c[exp_inds]
            exp_s = s[exp_inds]
        else:
            exp_c = c
            exp_s = s
        
        #find the number of data points in each quadrant
        top_right_s = exp_s[(exp_s >= 6) & (exp_c >= 0.25)] 
        top_left_s = exp_s[(exp_s >= 6) & (exp_c < 0.25)]
        bot_right_s = exp_s[(exp_s < 6) & (exp_c >= 0.25)]
        bot_left_s = exp_s[(exp_s < 6) & (exp_c < 0.25)]
        
        #build quadrant colors
        top_right_color = (1-top_right_s.size/float(exp_s.size),
                           1,
                           1-top_right_s.size/float(exp_s.size))
        top_left_color = (1-top_left_s.size/float(exp_s.size),
                           1,
                           1-top_left_s.size/float(exp_s.size))
        bot_right_color = (1-bot_right_s.size/float(exp_s.size),
                           1,
                           1-bot_right_s.size/float(exp_s.size))
        bot_left_color = (1-bot_left_s.size/float(exp_s.size),
                           1,
                           1-bot_left_s.size/float(exp_s.size))
        
        #build quadrant rectangles
        top_right_rect = patches.Rectangle((0.25,6),0.75,14, 
                                           color = top_right_color,
                                           zorder = 1)
        ax.add_patch(top_right_rect)
        top_left_rect = patches.Rectangle((-1,6),1.25,14, 
                                           color = top_left_color, 
                                           zorder = 1)
        ax.add_patch(top_left_rect)
        bot_right_rect = patches.Rectangle((0.25,0),0.75,6, 
                                           color = bot_right_color, 
                                           zorder = 1)
        ax.add_patch(bot_right_rect)
        bot_left_rect = patches.Rectangle((-1,0),1.25,6, 
                                           color = bot_left_color, 
                                           zorder = 1)
        ax.add_patch(bot_left_rect)
        
        #build the scatter plot and label the axis
        ax.scatter(exp_c,exp_s, zorder = 2, label='pert')
        if control_inds:
            ax.scatter(control_c,control_s,facecolor='r',zorder=3, label='control')
            plt.legend(loc='center left')
        ax.set_xlim(-1,1)
        ax.set_ylim(0,20)
        
        
        #add text annotations
        props = dict(boxstyle='round', facecolor='wheat')
        ax.text(0.05, 0.90, '%.0f%%' %(top_left_s.size/float(exp_s.size)*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        ax.text(0.85, 0.90, '%.0f%%' %(float(top_right_s.size)/exp_s.size*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        ax.text(0.05, 0.05, '%.0f%%' %(bot_left_s.size/float(exp_s.size)*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        ax.text(0.85, 0.05, '%.0f%%' %(bot_right_s.size/float(exp_s.size)*100),
                 transform=ax.transAxes,
                  fontsize=20,
                  bbox = props)
        
        #add cutoff annotation
        ax.axhline(y=s_thresh,linewidth=2,color='k')
        ax.axvline(x=c_tresh,linewidth=2,color='k')
        
        #add labels, add title, and increase tick font size
        plt.xlabel(xlabel, fontsize=20)
        plt.ylabel(ylabel, fontsize=20)
        if title:
            plt.title(title, fontsize=20)
        for tick in ax.xaxis.get_major_ticks():
            tick.label1.set_fontsize(20)
        for tick in ax.yaxis.get_major_ticks():
            tick.label1.set_fontsize(20)
        
        #show on screen unless out is specified.  If it is, write the figure to the path in out
        if not out:
            plt.show()
        else:
            plt.savefig(out)
